1607__STATIC_INLINE uint32_t LL_USART_GetBaudRate(USART_TypeDef *USARTx, uint32_t PeriphClk, uint32_t OverSampling)
1608{
1609 uint32_t usartdiv;
1610 uint32_t brrresult = 0x0U;
1611
1612 usartdiv = USARTx->BRR;
1613
1614 if (usartdiv == 0U)
1615 {
1616 /* Do not perform a division by 0 */
1617 }
1618 else if (OverSampling == LL_USART_OVERSAMPLING_8)
1619 {
1620 usartdiv = (uint16_t)((usartdiv & 0xFFF0U) | ((usartdiv & 0x0007U) << 1U)) ;
1621 if (usartdiv != 0U)
1622 {
1623 brrresult = (PeriphClk * 2U) / usartdiv;
1624 }
1625 }
1626 else
1627 {
1628 if ((usartdiv & 0xFFFFU) != 0U)

1632 }
